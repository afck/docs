fann_sys::fann_create_standard_array [−] [src]

pub unsafe extern fn fann_create_standard_array(num_layers: c_uint, layers: *const c_uint) -> *mut fann

Just like fann_create_standard, but with an array of layer sizes instead of individual parameters.

Example

// Creating an ANN with 2 input neurons, 1 output neuron,
// and two hidden layers with 8 and 9 neurons
let layers = [2, 8, 9, 1];
unsafe {
    let ann = fann_sys::fann_create_standard_array(4, layers.as_ptr());
}
